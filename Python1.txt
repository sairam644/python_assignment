def perfect(num):
    sum=0
    for i in range(1,num):
        if num%i==0:
            sum+=i
    if sum==num:
        return f"it is perfect number"
    else:
        return f"not a perfect number"
print (perfect(28))

def elementwise_greater_than(mylist,threshold):
    result=[]
    for i in range(len(mylist)):
        if mylist[i] > threshold:
            result.append(True)
        else:
            result.append(False)
    return result
print(elementwise_greater_than([1,2,3,4,5],3))     output:[false,false,false,true,true]

        return False
